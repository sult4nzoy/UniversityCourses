int main()

        std::cout << "3. Sök efter en kontakt." << std::endl;
        std::cout << "4. Avsluta program." << std::endl;
        std::cout << std::endl;
        std::cout << "Val: ";

        std::cin >> användarensVal;

        switch (användarensVal) //Använder switch istället för massa if- & else-satser.

        case 1:
        {

                {

                else
                {

            }

            while (true)
            {
                std::cout << "Efternamn på personen: "; // Användaren får skriva efternamn
                std::cin >> efternamn;

                if (!bokstäverKontroll(efternamn)) //Anropar funktionen för kontroll
                {
                    std::cout << "Efternamnet får endast innehålla bokstäver. Försök igen." << std::endl; //Felmeddelande

            std::vector<std::string> behållKontakt; // Här ska vi lagra de kontakter som inte ska tas bort

            {

                    if (temporärVek[1] == förnamnTaBort)

                        letaren = true; // Kontakt hittades, vi ska inte lägga till den i den nya listan
                    }
                    else
                    {
                        behållKontakt.insert(behållKontakt.end(), temporärVek.begin(), temporärVek.end());

                    temporärVek.clear(); // Rensa temporär vektor
                }
            }

                std::ofstream outFile("Kontaktbok", std::ios::trunc); // Skriv tillbaka till filen de kontakter som inte ska tas bort
